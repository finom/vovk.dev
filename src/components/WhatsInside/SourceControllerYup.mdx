```ts
import { prefix, patch } from 'vovk';
import { withYup } from 'vovk-yup';
import { UserSchema } from '@/yup-schemas';
import UserService from './UserService';

@prefix('users')
export default class UserController {
  @patch(':id')
  static updateUser = withYup(UserSchema.omit(['id']), async (req, { id }: { id: string }) =>
    UserService.updateUser(id, await req.json())
  );
}
```
